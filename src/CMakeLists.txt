SET( APCB_VALA_FILES
	Couchbase.vala
	CouchbasePlugin.vala
	CouchbaseSession.vala
	StorableCouchbase.vala
)

vala_precompile(APCB_VALA_C
	${APCB_VALA_FILES}
PACKAGES
	${APCB_VALA_PACKAGES}
OPTIONS
	${DEFAULT_VALA_OPTIONS}
GENERATE_VAPI
	${PPKGNAME}-${PPKGVERSION}
GENERATE_HEADER
	${PPKGNAME}-${PPKGVERSION})

set(APCB_LIB_PATHS
	${APCB_DEPS_LIBRARY_DIRS}
)
link_directories(${APCB_LIB_PATHS})
add_definitions(${APCB_DEPS_CFLAGS_ALL})

add_library(${PPKGNAME}-${PPKGVERSION} SHARED ${APCB_VALA_C})
target_link_libraries(${PPKGNAME}-${PPKGVERSION} ${APCB_DEPS_LIBRARIES})

install (
	FILES ${CMAKE_CURRENT_BINARY_DIR}/${PPKGNAME}-${PPKGVERSION}.vapi ${PPKGNAME}-${PPKGVERSION}.deps
	DESTINATION ${CMAKE_INSTALL_PREFIX}/share/ambition-framework/plugins/${PLUGINNAME}-${PPKGVERSION}/)
install (
	FILES ${CMAKE_CURRENT_BINARY_DIR}/${PPKGNAME}-${PPKGVERSION}.h
	DESTINATION ${CMAKE_INSTALL_PREFIX}/share/ambition-framework/plugins/${PLUGINNAME}-${PPKGVERSION}/)
install (
	TARGETS ${PPKGNAME}-${PPKGVERSION}
	DESTINATION ${CMAKE_INSTALL_PREFIX}/share/ambition-framework/plugins/${PLUGINNAME}-${PPKGVERSION}/)
