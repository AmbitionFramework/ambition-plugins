SET( AES_VALA_FILES
	SCGI.vala
	SCGIPlugin.vala
	SCGIUtility.vala
)
vala_precompile(AES_VALA_C
	${AES_VALA_FILES}
PACKAGES
    ${AES_VALA_PACKAGES}
OPTIONS
	${DEFAULT_VALA_OPTIONS}
GENERATE_VAPI
	${PPKGNAME}-${PPKGVERSION}
GENERATE_HEADER
	${PPKGNAME}-${PPKGVERSION}
)

link_directories(${AES_LIB_PATHS})
add_definitions(${AES_DEPS_CFLAGS_ALL})

add_library(${PPKGNAME}-${PPKGVERSION} SHARED ${AES_VALA_C})
target_link_libraries(${PPKGNAME}-${PPKGVERSION} ${AES_DEPS_LIBRARIES})

install (
	FILES ${CMAKE_CURRENT_BINARY_DIR}/${PPKGNAME}-${PPKGVERSION}.vapi
	      ${PPKGNAME}-${PPKGVERSION}.deps
	      ${CMAKE_CURRENT_BINARY_DIR}/${PPKGNAME}-${PPKGVERSION}.h
	DESTINATION ${CMAKE_INSTALL_PREFIX}
)
install (
	TARGETS ${PPKGNAME}-${PPKGVERSION}
	DESTINATION ${CMAKE_INSTALL_PREFIX}
)
